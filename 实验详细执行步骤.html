📋 9.1.1 基于 Kylin OS 的进程调度与优先级实验
C 语言pthreadnice/renicetasksetCFS
实验目的

通过编写高 CPU 负载的 pthread 多线程程序，观察 CFS 在不同 nice 值下的 CPU 时间分配现象。

详细执行步骤
1
创建并编译 pthread 程序
mkdir -p ~/labs/cfs_experiment
cd ~/labs/cfs_experiment
cat > nice-exp.c <<'EOF'
#include <stdio.h>
#include <pthread.h>
#include <sys/types.h>
#include <unistd.h>

void *thread_fun(void *param)
{
    printf("thread pid:%d, tid:%lu\n", getpid(), pthread_self());
    while (1) ;
    return NULL;
}

int main(void)
{
    pthread_t tid;
    int ret;
    printf("main pid:%d, tid:%lu\n", getpid(), pthread_self());
    ret = pthread_create(&tid, NULL, thread_fun, NULL);
    if (ret == -1) {
        perror("cannot create new thread");
        return 1;
    }
    if (pthread_join(tid, NULL) != 0) {
        perror("call pthread_join function fail");
        return 1;
    }
    return 0;
}
EOF
gcc nice-exp.c -o nice-exp -pthread
ls -l nice-exp
截图：能看到 nice-exp 文件生成，编译命令包含 -pthread。
2
启动两个进程并绑定同一 CPU
cd ~/labs/cfs_experiment
pkill nice-exp 2>/dev/null

taskset -c 0 ./nice-exp >/dev/null &
PID_A=$!

taskset -c 0 ./nice-exp >/dev/null &
PID_B=$!

echo "PID_A=$PID_A PID_B=$PID_B"
3
确认两个进程处于同一核心
taskset -cp $PID_A
taskset -cp $PID_B
ps -o pid,ni,psr,pcpu,comm -p $PID_A,$PID_B
4
提高其中一个进程优先级
sudo renice -n -5 -p $PID_A
sleep 8
ps -o pid,ni,psr,pcpu,comm -p $PID_A,$PID_B
确认 PID_A 的 NI 为 -5 后再截图。
5
使用 top 观察 CPU 占比
top -p $PID_A,$PID_B
进入 top 后等待 5 到 10 秒。预期现象是 NI=-5 的进程占用更多 CPU，NI=0 的进程占用更少 CPU。
6
结束实验进程
kill $PID_A $PID_B
pkill nice-exp 2>/dev/null
预期现象汇总
阶段预期现象原理
绑定同一 CPU两个 nice-exp 进程 PSR 相同，NI 均为 0，CPU 占比接近同一核心上权重相同，CFS 近似公平分配
renice 后PID_A 的 NI 为 -5，CPU 占比更高nice 值变小，调度权重提高
top 观察NI=-5 的进程占用更多 CPU，NI=0 的进程占用更少 CPUCFS 根据权重分配运行时间
注意事项
运行死循环程序会导致 CPU 占用率升高，实验结束后务必使用 kill $PID_A $PID_B 或 pkill nice-exp 清理进程。
renice -n -5 通常需要 sudo 权限。如果 top 中占比短时间内仍接近，等待几秒后再截图。
📋 9.1.2 实现 O(1) 调度器算法
C 语言位运算查表法实时调度
实验目的

通过查表法（OSUnMapTbl）和位运算实现 O(1) 复杂度调度器，验证无需循环即可找到最高优先级任务。

详细执行步骤
1
创建实验目录
mkdir -p ~/labs/scheduler_experiment
cd ~/labs/scheduler_experiment
2
编写 scheduler.c
nano scheduler.c
输入以下代码：
#include <stdio.h>

// 定义无符号8位整型
typedef unsigned char INT8U;

// 优先级查找表：给定8位二进制数，返回从低位到高位第一个为1的位的位置(0-7)
INT8U const OSUnMapTbl[256] = {
    0, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    6, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    7, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    6, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
    4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0
};

// 就绪组和就绪表
INT8U OSRdyGrp = 0;
INT8U OSRdyTbl[8] = {0};

// 辅助函数：将某个优先级的任务设为就绪状态
void make_task_ready(INT8U prio) {
    OSRdyGrp |= (1 << (prio >> 3));        // 设置组位
    OSRdyTbl[prio >> 3] |= (1 << (prio & 0x07)); // 设置表位
}

int main() {
    INT8U y, x, highest_prio;
    
    // 假设三个任务就绪，优先级分别是 45, 12, 27
    make_task_ready(45);
    make_task_ready(12);
    make_task_ready(27);
    
    // 核心算法：通过查表快速找到最高优先级
    y = OSUnMapTbl[OSRdyGrp];        // 找到有任务就绪的最高优先级组
    x = OSUnMapTbl[OSRdyTbl[y]];     // 在该组中找到最高优先级的具体位置
    highest_prio = (y << 3) + x;     // 拼接得到真实最高优先级
    
    printf("Ready tasks added: 45, 12, 27\n");
    printf("The highest priority ready task is: %d\n", highest_prio);
    return 0;
}
3
编译并运行
gcc scheduler.c -o scheduler
./scheduler
4
验证结果
预期输出：
Ready tasks added: 45, 12, 27
The highest priority ready task is: 12
结果分析：通过查表法和位运算，在没有使用循环遍历的情况下，瞬间锁定了最高优先级任务（12）。这体现了 O(1) 算法在实时操作系统中高效且执行时间可预测的优势。
原理图解
变量/操作含义本例计算
prio >> 3优先级所在组号（y）12 >> 3 = 1
prio & 0x07优先级在组内位置（x）12 & 7 = 4
OSRdyGrp哪些组有就绪任务bit1=1, bit5=1 → 0x22
OSUnMapTbl[OSRdyGrp]查表得最低有就绪位的组OSUnMapTbl[0x22] = 1
OSUnMapTbl[OSRdyTbl[1]]查表得组内最低就绪位OSRdyTbl[1] = 0x10 → OSUnMapTbl[0x10] = 4
highest_prio = (y<<3)+x组合得到最高优先级(1<<3)+4 = 12

📋 9.2.3 Slab 内存分配实验
内核模块Slab 分配器Linux Kernel
实验目的

使用内核 Slab 分配器进行内存分配与释放，掌握 kmem_cache_create / kmem_cache_alloc / kmem_cache_free 的用法。

详细执行步骤
1
创建实验目录
mkdir -p ~/labs/slab_experiment
cd ~/labs/slab_experiment
2
编写 slab_test.c
nano slab_test.c
#include <linux/module.h>
#include <linux/version.h>
#include <linux/init.h>
#include <linux/kernel.h>
#include <linux/types.h>
#include <linux/list.h>
#include <linux/slab.h>

#define ALLOC_SIZE 8

struct kmem_cache *kcp;

// 定义包含缓冲区和链表节点的数据结构
struct slab_test_buf {
    char buf[64];
    struct list_head list;
};

struct list_head stp_head;

static void print_stp(void)
{
    struct slab_test_buf *stp = NULL;
    list_for_each_entry(stp, &stp_head, list) {
        printk("%s\n", stp->buf);
    }
}

static void free_stp(void)
{
    struct slab_test_buf *stp = NULL;
    struct slab_test_buf *next = NULL;
    // 使用 safe 遍历以安全释放内存
    list_for_each_entry_safe(stp, next, &stp_head, list) {
        list_del(&stp->list);
        kmem_cache_free(kcp, stp);
    }
}

static int __init slab_test_init(void)
{
    int i;
    struct slab_test_buf *stp = NULL;
    
    printk("slab test module init\n");
    INIT_LIST_HEAD(&stp_head);
    
    // 创建 Slab 缓存
    kcp = kmem_cache_create("slab_test_buf", sizeof(struct slab_test_buf), 0, SLAB_HWCACHE_ALIGN, NULL);
    if (!kcp) {
        printk("kmem_cache_create failed\n");
        return -ENOMEM;
    }
    
    // 分配并初始化多个对象
    for (i = 0; i < ALLOC_SIZE; i++) {
        stp = kmem_cache_alloc(kcp, GFP_KERNEL);
        if (!stp) {
            printk("kmem_cache_alloc failed\n");
            free_stp();
            kmem_cache_destroy(kcp);
            return -ENOMEM;
        }
        sprintf(stp->buf, "hello slab %d", i);
        list_add(&stp->list, &stp_head);
    }
    
    printk("Allocated %d objects:\n", ALLOC_SIZE);
    print_stp();
    return 0;
}

static void __exit slab_test_exit(void)
{
    printk("slab test module exit\n");
    free_stp();
    kmem_cache_destroy(kcp);
}

module_init(slab_test_init);
module_exit(slab_test_exit);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Student");
MODULE_DESCRIPTION("Slab allocator test module");
3

all:
	make -C /lib/modules/$(shell uname -r)/build M=$(PWD) modules

clean:
	make -C /lib/modules/$(shell uname -r)/build M=$(PWD) clean

📋 9.2.4 内存回收实验（OOM Killer）
内存管理OOM Killerdmesg
实验目的

通过用户态程序疯狂申请物理内存，触发内核 OOM Killer，观察系统的内存回收机制。

详细执行步骤
1
关闭 Swap（切断系统"后路"）
为了让实验更快触发 OOM，先关闭虚拟内存：
sudo swapon --show
如有输出，表示已开启 Swap：
sudo swapoff -a
swapoff 是临时关闭，重启后会恢复。实验结束后可执行 sudo swapon -a 重新开启。
2
创建实验目录并编写 oom.c
mkdir -p ~/labs/oom_experiment
cd ~/labs/oom_experiment
nano oom.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

#define PAGE_SIZE 4096

int main() {
    int i = 0;
    char *p;
    
    while (1) {
        p = malloc(PAGE_SIZE);
        if (!p) {
            printf("malloc failed at %d pages\n", i);
            break;
        }
        // 写入数据，确保内存真正被分配（非 overcommit）
        memset(p, 0xAB, PAGE_SIZE);
        i++;
        
        if (i % 1000 == 0) {
            printf("Allocated %d pages (%d MB)\n", i, i * PAGE_SIZE / 1024 / 1024);
        }
    }
    
    printf("Total allocated: %d pages, sleeping...\n", i);
    while (1) sleep(1);
    return 0;
}

sudo yum install -y fuse fuse-devel
2
编写简易 FUSE 文件系统
核心需要实现以下回调函数：
static struct fuse_operations myfs_ops = {
    .getattr = myfs_getattr,    // 获取文件属性
    .readdir = myfs_readdir,    // 读取目录
    .open    = myfs_open,       // 打开文件
    .read    = myfs_read,       // 读取文件
    .write   = myfs_write,      // 写入文件
    .create  = myfs_create,    // 创建文件
    .mkdir   = myfs_mkdir,     // 创建目录
};
文件数据结构（内存存储）：
struct myfs_node {
    char name[256];
    char *data;
    size_t size;
    int is_dir;
    struct myfs_node *children;
    int child_count;
};
3
编译与挂载
gcc -Wall myfs.c `pkg-config --cflags --libs fuse` -o myfs
mkdir -p /tmp/myfs_mount
./myfs /tmp/myfs_mount
4
测试文件操作
echo "hello" > /tmp/myfs_mount/test.txt
cat /tmp/myfs_mount/test.txt
ls /tmp/myfs_mount/
推荐实现路径 B：进阶 Page Cache（基于内核模块）
5
Page Cache 核心思路
在内核文件系统中使用 address_space 结构管理页缓存：
struct address_space *mapping = inode->i_mapping;
struct page *page = find_get_page(mapping, index);  // 查找缓存页
if (!page) {
    page = read_cache_page(mapping, index, filler, data);  // 读取到缓存
}
Page Cache 可显著提升重复读取性能，通过 free -m 或 cat /proc/meminfo | grep Cached 观察缓存命中效果。

9.6.1 容器化负载均衡部署实践
1
部署 Docker 服务
sudo yum install -y docker
sudo systemctl restart docker
sudo systemctl enable docker
sudo systemctl status docker
按 "q" 退出 status 查看。
2
导入基础镜像
sudo docker pull cr.kylinos.cn/kylin/kylin-server-init:v10sp1
sudo docker tag cr.kylinos.cn/kylin/kylin-server-init:v10sp1 kylin-base:v10sp1
sudo docker run --rm -it kylin-base:v10sp1 uname -m
应显示 x86_64。验证成功后查看镜像：sudo docker images
3
编写 Dockerfile 构建 Nginx
创建 Dockerfile 目录：
mkdir ~/docker-nginx && cd ~/docker-nginx
vim Dockerfile
FROM kylin-base:v10sp1
RUN yum -y install gcc make pcre-devel zlib-devel tar zlib
ADD nginx-1.15.2.tar.gz /usr/src/
RUN cd /usr/src/nginx-1.15.2 && mkdir -p /usr/local/nginx && \
    ./configure --prefix=/usr/local/nginx && \
    make && make install && \
    ln -s /usr/local/nginx/sbin/nginx /usr/local/sbin/
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
构建镜像：
sudo docker build -t nginx-kylin:v1 .
4
构建 Tomcat 镜像
类似地，编写 Dockerfile 安装 JDK 和 Tomcat，构建 tomcat-kylin:v1 镜像。
5
创建容器
# 创建 Nginx 容器（负载均衡器）
sudo docker run -d --name nginx-lb -p 80:80 nginx-kylin:v1

# 创建两个 Tomcat 容器（后端服务）
sudo docker run -d --name tomcat-1 tomcat-kylin:v1
sudo docker run -d --name tomcat-2 tomcat-kylin:v1
6
配置 Nginx 负载均衡
进入 Nginx 容器，修改 nginx.conf：
sudo docker exec -it nginx-lb /bin/bash
vim /usr/local/nginx/conf/nginx.conf
http {
    upstream tomcat_backend {
        server tomcat-1:8080;
        server tomcat-2:8080;
    }
    server {
        listen 80;
        location / {
            proxy_pass http://tomcat_backend;
        }
    }
}
重启 Nginx：
/usr/local/nginx/sbin/nginx -s reload
7
验证负载均衡
在宿主机浏览器访问 http://宿主机IP，多次刷新，观察请求被分发到不同 Tomcat 容器。
也可使用 curl 命令多次请求：curl http://localhost
9.6.2 电商 Web 大并发解耦解决方案
1
Docker 安装与基础镜像导入
sudo yum install -y docker
sudo systemctl restart docker && sudo systemctl enable docker
sudo docker pull cr.kylinos.cn/kylin/kylin-server-init:v10sp1
sudo docker tag cr.kylinos.cn/kylin/kylin-server-init:v10sp1 kylin-base:v10sp1
2
创建 nginx-1 和 nginx-2 容器（带持久化存储）
sudo mkdir -p /data/storage/docker/{nginx_1,nginx_2}/{conf,html}

sudo docker run -it -p 81:80 -d --name nginx-1 \
    -v /data/storage/docker/nginx_1/conf:/usr/local/nginx/conf \
    -v /data/storage/docker/nginx_1/html:/usr/local/nginx/html \
    kylin-base:v10sp1

sudo docker run -it -p 82:80 -d --name nginx-2 \
    -v /data/storage/docker/nginx_2/conf:/usr/local/nginx/conf \
    -v /data/storage/docker/nginx_2/html:/usr/local/nginx/html \
    kylin-base:v10sp1
3
为两个容器配置 LNMP 环境
分别进入 nginx-1 和 nginx-2 容器，安装 Nginx、MySQL、PHP：
sudo docker exec -it nginx-1 /bin/bash
yum install -y nginx mariadb-server php php-fpm
# 配置 nginx.conf 支持 PHP
# 启动服务：systemctl start nginx mariadb php-fpm
对 nginx-2 重复相同操作。
4
在本机部署 Nginx 集群（轮询）
在宿主机（或另一个容器）安装 Nginx，配置 upstream：
upstream backend {
    server 宿主机IP:81;
    server 宿主机IP:82;
}

server {
    listen 80;
    location / {
        proxy_pass http://backend;
    }
}
默认使用轮询算法（round-robin），请求依次分发到 nginx-1 和 nginx-2。
5
验证集群
在 nginx-1 和 nginx-2 的 html 目录中放置不同内容的 index.html，通过浏览器访问宿主机 80 端口，多次刷新，确认轮询效果。
